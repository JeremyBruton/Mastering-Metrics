4.3 Now try Stargazer

x <- sample(1:5,100,TRUE)
z <- sample(5:10,100,TRUE)
y <- sample(1:10,100,TRUE)
df <- data.frame(x,y,z)
reg1 <- lm(y ~ x ,data=df)
reg2 <- lm(y ~ z,data=df)
reg3 <- lm(y ~ x + z,data=df)
